```python
def draw_spider_chart(ax, scores, labels):
    """
    scores: dict {'RD': 31, 'SR': 94, ...}
    labels: list в нужном порядке ['RD', 'SR', 'AD', 'FL', 'RC', 'EN', 'BL']
    """
    n = len(labels)
    
    # Углы для каждой оси (начиная с верха, по часовой стрелке)
    angles = [np.pi/2 - i * 2*np.pi/n for i in range(n)]
    angles.append(angles[0])  # замыкаем контур
    
    # Значения в том же порядке
    values = [scores[label] / 100 for label in labels]  # нормализация к 0-1
    values.append(values[0])  # замыкаем контур
    
    # Рисуем фоновую сетку (концентрические круги)
    for level in [0.25, 0.5, 0.75, 1.0]:
        circle_x = [level * np.cos(a) for a in angles]
        circle_y = [level * np.sin(a) for a in angles]
        ax.plot(circle_x, circle_y, color='#3a3a5c', linewidth=0.5, alpha=0.5)
    
    # Рисуем оси
    for angle in angles[:-1]:
        ax.plot([0, np.cos(angle)], [0, np.sin(angle)], 
                color='#3a3a5c', linewidth=0.5, alpha=0.5)
    
    # Рисуем данные
    data_x = [v * np.cos(a) for v, a in zip(values, angles)]
    data_y = [v * np.sin(a) for v, a in zip(values, angles)]
    
    ax.fill(data_x, data_y, color='#4ecdc4', alpha=0.3)
    ax.plot(data_x, data_y, color='#4ecdc4', linewidth=2)
    
    # Точки на вершинах
    for x, y in zip(data_x[:-1], data_y[:-1]):
        ax.scatter(x, y, color='#4ecdc4', s=50, zorder=5)
    
    # Подписи осей (снаружи)
    label_offset = 1.15
    for i, label in enumerate(labels):
        angle = angles[i]
        x = label_offset * np.cos(angle)
        y = label_offset * np.sin(angle)
        
        # Подпись оси
        ax.text(x, y, label, ha='center', va='center', 
                fontsize=10, color='#ffffff', fontweight='bold')
        
        # Процент рядом с подписью (чуть ближе к центру)
        pct_offset = 1.0
        px = pct_offset * np.cos(angle)
        py = pct_offset * np.sin(angle)
        ax.text(px, py - 0.1, f"{scores[label]}%", ha='center', va='top',
                fontsize=9, color='#4ecdc4')
    
    ax.set_xlim(-1.4, 1.4)
    ax.set_ylim(-1.4, 1.4)
    ax.set_aspect('equal')
    ax.axis('off')
```

```python
def calculate_biological_age(rmssd_ms, sdnn_ms, stress_index):
    """
    ВАЖНО: rmssd_ms и sdnn_ms должны быть в МИЛЛИСЕКУНДАХ!
    """
    # Проверка входных данных
    print(f"DEBUG bio_age: rmssd={rmssd_ms}, sdnn={sdnn_ms}, si={stress_index}")
    
    # Базовая оценка через RMSSD
    rmssd = rmssd_ms  # убедиться что в мс
    
    if rmssd >= 50:
        base_age = 20
    elif rmssd >= 40:
        base_age = 20 + (50 - rmssd) * 1.0
    elif rmssd >= 30:
        base_age = 30 + (40 - rmssd) * 1.5
    elif rmssd >= 20:
        base_age = 45 + (30 - rmssd) * 1.5
    elif rmssd >= 10:
        base_age = 60 + (20 - rmssd) * 1.5
    else:
        base_age = 75 + (10 - rmssd) * 2.0
    
    # Коррекция по SDNN
    if sdnn_ms > 100:
        base_age -= 5
    elif sdnn_ms < 30:
        base_age += 5
    
    # Коррекция по Stress Index
    if stress_index < 100:
        base_age -= 3
    elif stress_index > 300:
        base_age += 5
    
    result = max(18, min(90, round(base_age)))
    print(f"DEBUG bio_age result: {result}")
```

```python
def stress_resistance_score(stress_index, lf_hf_ratio=None):
    """
    Стрессоустойчивость с опциональной коррекцией по LF/HF
    """
    # Базовый расчёт из SI
    if stress_index <= 50:
        sr_base = 100
    elif stress_index <= 150:
        sr_base = 100 - (stress_index - 50) * 0.25
    elif stress_index <= 300:
        sr_base = 75 - (stress_index - 150) * 0.33
    elif stress_index <= 500:
        sr_base = 25 - (stress_index - 300) * 0.10
    else:
        sr_base = max(0, 5 - (stress_index - 500) * 0.01)
    
    # Коррекция по LF/HF (если передан)
    if lf_hf_ratio is not None and lf_hf_ratio > 4.0:
        penalty = min(40, (lf_hf_ratio - 4) * 6)  # до -40%
        sr_base = max(10, sr_base - penalty)
    
    return round(sr_base)

# Пример:
# SI = 75.9, LF/HF = 10.7
# sr_base = 100 - (75.9-50)*0.25 = 93.5%
# penalty = min(40, (10.7-4)*6) = min(40, 40.2) = 40
# SR = max(10, 93.5 - 40) = 53.5%
```

```python
def analyze_trend(hr_data):
    """
    Анализ тренда HR
    """
    # Тренд
    q1 = np.mean(hr_data[:len(hr_data)//4])
    q4 = np.mean(hr_data[-len(hr_data)//4:])
    
    if q4 > q1 * 1.1:
        trend = 'нарастающий'
    elif q4 < q1 * 0.9:
        trend = 'снижающийся'
    else:
        trend = 'стабильный'
    
    # Вариабельность
    std = np.std(hr_data)
    if std < 3:
        variability = 'низкая'
    elif std < 6:
        variability = 'умеренная'
    else:
        variability = 'высокая'
    
    return {
        'trend': trend,
        'variability': variability,
        'std': round(std, 1)
    }
```
